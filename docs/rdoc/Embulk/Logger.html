def initialize(*args)
  if args.length == 1
    a = args[0]
    if a.is_a?(Adapter)
      @logger = a
    elsif a.is_a?(::Logger)
      @logger = StandardLoggerAdapter.new(a)
    elsif RUBY_PLATFORM =~ /java/i && (org.slf4j.Logger rescue nil) && a.is_a?(org.slf4j.Logger)
      @logger = Slf4jAdapter.new(a)
    else
      @logger = StandardLoggerAdapter.new(*args)
    end
  else
    @logger = StandardLoggerAdapter.new(*args)
  end
end

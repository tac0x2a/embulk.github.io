def flush
  unless @buffer.empty?
    @java_file_output.add(@buffer.to_java)
    @buffer.clear
  end
  nil
end

def write(buffer)
  buffer.force_encoding('ASCII-8BIT')  # TODO this is destructively change buffer
  @buffer << buffer
  if @buffer.size > @flush_size
    flush
  end
  nil
end

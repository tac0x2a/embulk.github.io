def self.new_java_type(ruby_type)
  case ruby_type
  when :boolean
    Java::Types::BOOLEAN
  when :long
    Java::Types::LONG
  when :double
    Java::Types::DOUBLE
  when :string
    Java::Types::STRING
  when :timestamp
    Java::Types::TIMESTAMP
  else
    raise ArgumentError, "Unknown type #{ruby_type.inspect}: supported types are :boolean, :long, :double, :string and :timestamp"
  end
end
